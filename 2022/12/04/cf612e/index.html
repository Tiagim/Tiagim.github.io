CF612E Square Root of Permutation

2022/12/4 oi 题解

简要题面

给定一个的排列，求一个排列，使得对于任意，。无解输出。

。

输入格式

第一行一个数。
第二行个数，表示排列。

输出格式

一行个数，表示排列。

样例1

样例输入1

4
2 1 4 3

样例输出1

3 4 2 1

样例2

样例输入2

4
2 1 3 4

样例输出2

-1

样例3

样例输入3

5
2 3 4 5 1

样例输出3

4 5 1 2 3

题解

既然置换是，所以就想到了建图。

进一步发现这张图是由几个环组成的（因为，这是一个排列，假设能转到，那么一定有一个能转到），而中的奇环在中也是由同样的数组成的奇环，中的偶环会变成两个大小相等的偶环。

于是对建图。找出对应的环奇环单独处理，两个大小相同的偶环一起处理。就能还原出对应的图了。

#include<bits/stdc++.h>
using namespace std;
int n,p[1000005],vis[1000005],ans[1000005],num;
vector<int>a[1000005],b[1000005];
int main(){
    freopen("deepdarkfantasy.in","r",stdin);
    freopen("deepdarkfantasy.out","w",stdout);
    scanf("%d",&n);
    for(register int i(1);i<=n;++i)scanf("%d",p+i);
    for(register int i(1);i<=n;++i){
        num=0;
        for(register int j=i;!vis[j];j=p[j])vis[j]=1,++num,a[i].push_back(j);
        b[num].push_back(i);
    }
    for(register int i(1);i<=n;++i)if(!(i&1)&&(b[i].size()&1)){puts("-1");return 0;}
    memset(vis,0,sizeof(vis));
    for(register int i(0);i<=(int)b[1].size()-1;++i)ans[b[1][i]]=b[1][i]; 
    for(register int i(2);i<=n;++i){
        for(register int j(0);j<=(int)b[i].size()-1;++j){
            if(i&1){
                int x=a[b[i][j]][0],y=a[b[i][j]][i/2+1];
                while(!vis[x])vis[x]=vis[y]=1,ans[x]=y,x=p[x],ans[y]=x,y=p[y];
            }
            else{
                int x=a[b[i][j]][0],y=a[b[i][j+1]][0];
                while(!vis[x]&&!vis[y])vis[x]=vis[y]=1,ans[x]=y,x=p[x],ans[y]=x,y=p[y];++j;
            }
        }
    }
    for(register int i(1);i<=n;++i)printf("%d ",ans[i]);
    return 0;
}
